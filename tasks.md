### T1.1 — Create xrplClient.js

**File:** `apps/web/lib/xrplClient.js`

Singleton xrpl.Client for devnet. Used by all hooks.

```javascript
import { Client } from "xrpl"
import { DEFAULT_NETWORK } from "./networks"

let client = null

export async function getClient() {
  if (!client || !client.isConnected()) {
    client = new Client(DEFAULT_NETWORK.wss)
    await client.connect()
  }
  return client
}

export async function disconnectClient() {
  if (client) await client.disconnect()
  client = null
}
```

### T1.3 — Create useVault hook

**File:** `apps/web/hooks/useVault.js`

Functions:
- `createVault(asset, metadata)` → submits `VaultCreate` tx, returns `vaultId`
- `deposit(vaultId, amount)` → submits `VaultDeposit` tx
- `withdraw(vaultId, amount)` → submits `VaultWithdraw` tx
- `getVaultInfo(vaultId)` → calls `vault_info` RPC, returns `{ totalAssets, totalShares, sharePrice }`
- `getShareBalance(vaultId, account)` → reads MPToken balance for vault share MPT

All transactions signed via `walletManager` from WalletProvider.

### T1.4 — Create useLoan hook

**File:** `apps/web/hooks/useLoan.js`

Functions:
- `createLoanBroker(vaultId, managementFeeRate)` → `LoanBrokerSet` tx, returns `loanBrokerId`
- `depositCover(loanBrokerId, amount)` → `LoanBrokerCoverDeposit` tx
- `createLoan(loanBrokerId, borrowerAddress, principal, interestRate, paymentTotal, paymentInterval, gracePeriod)` → `LoanSet` tx (cosigned: broker signs first, borrower signs second via `xrpl.signLoanSetByCounterparty`)
- `payLoan(loanId, amount, flags)` → `LoanPay` tx (flags: `tfLoanFullPayment`, `tfLoanLatePayment`)
- `manageLoan(loanId, action)` → `LoanManage` tx (action: `"impair"` | `"default"` | `"unimpair"`)
- `deleteLoan(loanId)` → `LoanDelete` tx
- `getLoanInfo(loanId)` → `ledger_entry` RPC for Loan object

### T1.5 — Create useEscrow hook

**File:** `apps/web/hooks/useEscrow.js`

Functions:
- `createEscrow(destination, amount, condition, cancelAfter)` → `EscrowCreate` tx, returns `{ escrowId, sequence }`
  - `condition`: hex-encoded SHA-256 hash
  - `cancelAfter`: Ripple epoch timestamp
- `finishEscrow(owner, sequence, fulfillment)` → `EscrowFinish` tx
  - `fulfillment`: hex-encoded preimage
- `cancelEscrow(owner, sequence)` → `EscrowCancel` tx (only works after cancelAfter)
- `getEscrow(owner, sequence)` → `ledger_entry` RPC for Escrow object

### T1.6 — Create useTickets hook

**File:** `apps/web/hooks/useTickets.js`

Functions:
- `createTickets(count)` → `TicketCreate` tx with `TicketCount: count`, returns `ticketSequences[]` (array of allocated sequence numbers, extracted from tx metadata)
- `buildPresignedOffers(ticketSequences, pair, amountPerBuy, side)` → builds N `OfferCreate` txs, each using a different `TicketSequence`, with `tfImmediateOrCancel` flag. Returns unsigned tx objects.
- `signAll(txs)` → signs all txs via walletManager. Returns signed blobs `string[]`.

### T3.5 — Create devnet-loop.js

**File:** `apps/watcher/src/devnet-loop.js`

DevnetLoop class:
- `start()` → subscribe to devnet `ledgerClosed` event
- `onLedger()`:
  1. `getPrice()` → `client.request({ command: "book_offers", ... })`, compute mid price
  2. For each active order in cache:
     - If TRAILING_STOP: update `highestPrice` and `computedTrigger` if new high
     - Check condition: `checkTrigger(order, currentPrice)` → boolean
     - If triggered: `executeOrder(order)`
  3. For each DCA schedule: if `Date.now() >= nextSubmitTime` → `submitNextDca(schedule)`

- `checkTrigger(order, price)`:
  ```javascript
  switch (order.orderType) {
    case "STOP_LOSS":  return order.side === "SELL" ? price <= order.triggerPrice : price >= order.triggerPrice
    case "TAKE_PROFIT": return order.side === "SELL" ? price >= order.triggerPrice : price <= order.triggerPrice
    case "TRAILING_STOP": return price <= order.computedTrigger
  }
  ```

- `executeOrder(order)`:
  1. `EscrowFinish` → tx: `{ TransactionType: "EscrowFinish", Owner: order.owner, OfferSequence: order.escrowSequence, Condition: order.condition, Fulfillment: order.preimage }`
  2. `OfferCreate` → tx: `{ TransactionType: "OfferCreate", TakerPays: ..., TakerGets: ..., Flags: 131072 }` (tfImmediateOrCancel)
  3. `Payment` → send proceeds back to `order.owner`
  4. Remove from cache

- `submitNextDca(schedule)`:
  1. Take next signed blob from `schedule.signedBlobs`
  2. `client.submit(blob)`
  3. `schedule.completed++`, update `nextSubmitTime`
  4. If `completed === total` → remove from cache

### T3.6 — Create dca-scheduler.js

**File:** `apps/watcher/src/dca-scheduler.js`

Separated DCA logic (called by devnet-loop):
```javascript
class DcaScheduler {
  constructor(client) { this.client = client }

  async submitNext(schedule) {
    const blob = schedule.signedBlobs[schedule.completed]
    const result = await this.client.submit(blob)
    schedule.completed++
    schedule.nextSubmitTime = Date.now() + schedule.intervalMs
    return result
  }

  isDue(schedule) {
    return Date.now() >= schedule.nextSubmitTime && schedule.completed < schedule.total
  }
}
```
